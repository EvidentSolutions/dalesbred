== Downloading

=== Gradle

[source,kotlin]
[subs="verbatim,attributes"]
----
repositories {
    mavenCentral()
}

dependencies {
    implementation("org.dalesbred:dalesbred:{revnumber}")
}
----

=== Maven

[source,xml]
[subs="verbatim,quotes,attributes"]
----
<dependency>
    <groupId>org.dalesbred</groupId>
    <artifactId>dalesbred</artifactId>
    <version>{revnumber}</version>
</dependency>
----
